<action>
Implement `src/main/entitlement/state.ts` — a pure (row, now) → state function. Take an `EntitlementRow` (raw row from DB) and a `Date`. Return one of 10 `EntitlementState` discriminated-union variants per RESEARCH §9:
```
type EntitlementState =
  | { kind: 'trial-active-quiet' | 'trial-active-day50' | 'trial-active-day55' | 'trial-active-day59'; daysRemaining: number; trialExpiresAt: string }
  | { kind: 'trial-expired-grace'; trialExpiresAt: string; hoursOfGraceRemaining: number }
  | { kind: 'trial-locked'; trialExpiresAt: string }
  | { kind: 'pro-active'; subscriptionUntil: string }
  | { kind: 'pro-grace'; lastVerifiedAt: string; daysUntilLock: number }
  | { kind: 'pro-locked'; lastVerifiedAt: string }
  | { kind: 'clock-skew-warn'; skewDays: number; underlyingState: EntitlementState };
```

Rules (compute from JWT claims when available, NOT local row's tier column directly — because tier can be tampered with):
- If no entitlement row exists → throw a programmer-error (caller must seed via trial-start first)
- If `tier === 'trial'`:
  - daysRemaining = ceil((trialExpiresAt - now) / 86400000)
  - daysRemaining > 10 → `trial-active-quiet`
  - daysRemaining == 10 → `trial-active-day50` (50 = 60-10)
  - daysRemaining == 5 → `trial-active-day55`
  - daysRemaining ∈ [1, 2] → `trial-active-day59`
  - daysRemaining == 0 && now < trialExpiresAt + 24h → `trial-expired-grace`
  - now >= trialExpiresAt → `trial-locked`
- If `tier === 'pro'`:
  - secondsSinceVerify = (now - last_verified_at) / 1000
  - secondsSinceVerify < 14d AND now < jwt.exp → `pro-active`
  - secondsSinceVerify between 24h and 14d → `pro-grace` (offline > 24h but inside grace)
  - secondsSinceVerify >= 14d → `pro-locked`
- Clock-skew check: if jwt.iat > (now + 7d epoch) or jwt.iat < (now - 7d epoch) AND row is otherwise active, wrap result in `clock-skew-warn`. Skew does NOT cause a lock.

Pure: no I/O, no DB, no clock other than the `now` argument.

```ts
export type EntitlementAction = 'email_send' | 'calendar_change' | 'task_push' | 'briefing_generate' | 'rag_ask';
export type EntitlementErrorCode = 'no-entitlement' | 'trial-locked' | 'pro-locked' | 'jwt-invalid' | 'jwt-expired';
export class EntitlementError extends Error { ... readonly code: EntitlementErrorCode; readonly action: EntitlementAction }

export async function assertEntitled(db: Db, action: EntitlementAction): Promise<void> {
  const row = db.prepare('SELECT ... FROM entitlement WHERE id = 1').get() as EntitlementRow | undefined;
  if (!row) throw new EntitlementError('no-entitlement', action, '...');
  if (!row.jwt) throw new EntitlementError('jwt-invalid', action, '...');
  let claims: EntitlementClaims;
  try {
    claims = await verifyEntitlementJwt(row.jwt);
  } catch (e) {
    throw new EntitlementError('jwt-invalid', action, '...');
  }
  // re-derive state from CLAIMS (not row.tier) — claims are signature-protected
  const state = computeEntitlementState({ ...row, /* override tier from claims */ tier: claims.tier, jwt_iat: new Date(claims.iat*1000).toISOString(), jwt_exp: new Date(claims.exp*1000).toISOString(), trial_started_at: claims.trial_started_at ?? row.trial_started_at, trial_expires_at: claims.trial_expires_at ?? row.trial_expires_at }, new Date());
  if (state.kind === 'trial-locked') throw new EntitlementError('trial-locked', action, '...');
  if (state.kind === 'pro-locked') throw new EntitlementError('pro-locked', action, '...');
  // trial-expired-grace, clock-skew-warn, and all -active/-grace states ALLOW the action
}
```
