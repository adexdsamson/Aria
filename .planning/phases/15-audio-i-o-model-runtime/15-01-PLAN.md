```typescript
// Keys namespaced with a prefix that cohabits cleanly with existing settings rows.
const KEY_PREFIX = 'backgroundActivity.';
function fullKey(key: BgPrefKey): string { return KEY_PREFIX + key; }
export function readBgPref(db: Db | null, key: BgPrefKey, fallback: boolean): boolean {
  if (!db) return fallback;                       // db-null tolerant (pre-unlock)
  try {
    const row = db.prepare('SELECT v FROM settings WHERE k = ?').get(fullKey(key)) as { v?: string } | undefined;
    return deserialise(row?.v, fallback);
  } catch { return fallback; }
}
export function writeBgPref(db: Db | null, key: BgPrefKey, value: boolean): void {
  if (!db) throw new Error('writeBgPref: db is null (vault sealed)');
  db.prepare(`INSERT INTO settings (k, v) VALUES (?, ?) ON CONFLICT(k) DO UPDATE SET v = excluded.v`).run(fullKey(key), serialise(value));
}
```
