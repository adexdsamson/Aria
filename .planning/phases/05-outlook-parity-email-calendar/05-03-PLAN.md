    ```tsx
    // RecurrenceUnsupportedPill.tsx
    export function RecurrenceUnsupportedPill({ event }: { event: CanonicalEvent }) {
      if (!event.recurrence_unsupported) return null;
      return (
        <a
          href={event.webLink}
          target="_blank"
          rel="noopener noreferrer"
          className="inline-flex items-center gap-1 bg-amber-100 text-amber-900 text-xs px-2 py-0.5 rounded"
          title="This recurrence pattern uses features Aria cannot fully represent locally. Click to view the full series in Outlook."
        >
          <Icon name="external-link" />
          complex recurrence — see in Outlook
        </a>
      );
    }

    // useRecurrenceUnsupportedToast.ts
    export function useRecurrenceUnsupportedToast(events: CanonicalEvent[], accountId: string) {
      useEffect(() => {
        if (!accountId) return;
        const key = `aria:recurrence-unsupported-toast-shown:${accountId}`;
        if (localStorage.getItem(key)) return;
        const hasUnsupported = events.some(e => e.recurrence_unsupported);
        if (!hasUnsupported) return;
        toast.info(
          "Some Outlook events use recurrence patterns Aria can't fully represent. They'll show with a 'View in Outlook' badge.",
          { duration: 8000 },
        );
        localStorage.setItem(key, new Date().toISOString());
      }, [events, accountId]);
    }
    ```

    ```ts
      if (backend === 'basic_text') {
        const accounts = listProviderAccounts(db).filter(r => r.provider_key === 'google');
        const accountIds = accounts.map(a => a.account_id);
        emitIpc('keyring:basic-text-skip', { accountIds });
        return {
          droppedCount: 0,
          failedCount: 0,
          skippedCount: accountIds.length,
          reason: 'basic_text-backend',
          accountIds,
        };
      }

      const rows = listProviderAccounts(db).filter(r => r.provider_key === 'google');
      const results = rows.map(r => dropLegacyGoogleEntry(r.account_id));
      const failures = results.filter(r => !r.dropped && r.reason !== 'no-legacy-entry');
      for (const f of failures) {
        markAccountStatus(db, { provider_key: 'google', account_id: f.accountId },
                          { last_error_message: `Legacy keyring entry retained — ${f.reason}` });
      }
      if (failures.length > 0) {
        emitIpc('keyring:legacy-retained', { accountIds: failures.map(f => f.accountId) });
      }
      return {
        droppedCount: results.filter(r => r.dropped).length,
        failedCount: failures.length,
        skippedCount: 0,
        failures,
      };
    }

    function dropLegacyGoogleEntry(accountId: string) {
      const legacy = readLegacyGoogleTokensEntry(accountId);
      const next   = getProviderTokens(`google:${accountId}`);
      if (!legacy) return { dropped: false, accountId, reason: 'no-legacy-entry' };
      if (!next)   return { dropped: false, accountId, reason: 'no-new-entry' };

      let nextEmail: string | undefined;
      try { nextEmail = decryptProviderTokens(next).email; }
      catch (err) { return { dropped: false, accountId, reason: `decrypt-failed: ${err instanceof Error ? err.message : String(err)}` }; }

      if (nextEmail !== accountId) {
        return { dropped: false, accountId, reason: `identity-mismatch: ${nextEmail} !== ${accountId}` };
      }

      deleteLegacyGoogleTokensEntry(accountId);
      return { dropped: true, accountId };
    }
    ```
