    **R4 — Per-account legacy keyring drop:** implement in src/main/secrets/safeStorage.ts as `dropLegacyGoogleEntry(accountId): Promise<{dropped: boolean, accountId: string, reason?: string}>`. Algorithm:

    ```
    function dropLegacyGoogleEntry(accountId) {
      const legacy = readLegacyGoogleTokensEntry(accountId);
      const next   = getProviderTokens(`google:${accountId}`);
      if (!legacy) return { dropped: false, accountId, reason: 'no-legacy-entry' };
      if (!next)   return { dropped: false, accountId, reason: 'no-new-entry' };

      // Verify the new entry decrypts + identity matches.
      let nextEmail;
      try { nextEmail = decryptProviderTokens(next).email; }
      catch (err) { return { dropped: false, accountId, reason: `decrypt-failed: ${err.message}` }; }

      if (nextEmail !== accountId) {
        return { dropped: false, accountId, reason: `identity-mismatch: ${nextEmail} !== ${accountId}` };
      }

      deleteLegacyGoogleTokensEntry(accountId);   // single-account scoped delete
      return { dropped: true, accountId };
    }

    function runDropLegacyGoogleKeyringPerAccount(db) {
      const rows = listProviderAccounts(db).filter(r => r.provider_key === 'google');
      const results = rows.map(r => dropLegacyGoogleEntry(r.account_id));
      const failures = results.filter(r => !r.dropped && r.reason !== 'no-legacy-entry');
      for (const f of failures) {
        markAccountStatus(db, { provider_key: 'google', account_id: f.accountId },
                          { last_error_message: `Legacy keyring entry retained — ${f.reason}` });
      }
      if (failures.length > 0) {
        emitIpc('keyring:legacy-retained', { accountIds: failures.map(f => f.accountId) });
      }
      return { droppedCount: results.filter(r => r.dropped).length,
               failedCount: failures.length, failures };
    }
    ```
