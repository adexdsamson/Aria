  approved → sending → failed            (email dispatch exception)
  approved → sent                        (calendar happy path)

    export async function recoverInflightSends(db: Db, emitBannerIpc?: (payload: any) => void) {
      const rows = db.prepare(
        `SELECT id, idempotency_key, provider_key, account_id FROM approval WHERE state='sending'`
      ).all() as Array<{ id: string; idempotency_key: string; provider_key: string; account_id: string }>;

      const stuck: Array<{ id: string; reason: string; idempotency_key: string }> = [];
      for (const row of rows) {
        try {
          const provider = getProvider(row.provider_key as ProviderKey, row.account_id);
          const found = await provider.mail!.findSentByIdempotencyKey(row.idempotency_key);
          if (found) {
            transitionTo(db, row.id, 'sent', { sent_at: new Date().toISOString() });
            logEmailSend(db, { approval_id: row.id, phase: 'sent', provider_message_id: found, recovered: true });
            continue;
          }
          // NOT FOUND → user decision.
          transitionTo(db, row.id, 'needs-operator-decision');
          db.prepare(`UPDATE approval SET last_error_message=? WHERE id=?`).run(
            `Aria couldn't confirm whether this message was sent — please check your Sent folder. Idempotency key: ${row.idempotency_key}`,
            row.id,
          );
          stuck.push({ id: row.id, reason: 'not-found-in-sent', idempotency_key: row.idempotency_key });
        } catch (err) {
          transitionTo(db, row.id, 'needs-operator-decision');
          db.prepare(`UPDATE approval SET last_error_message=? WHERE id=?`).run(
            `Aria couldn't verify send (${err instanceof Error ? err.message : String(err)}). Idempotency key: ${row.idempotency_key}`,
            row.id,
          );
