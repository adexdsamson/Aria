useReadAloudQueue.ts pattern (D-05/D-07 from RESEARCH.md Pattern 3 + PATTERNS.md):
```typescript
export function useReadAloudQueue(player: KokoroPlayerHandle, speed: number) {
  const queueRef = useRef<Promise<void>>(Promise.resolve());
  const enqueue = useCallback((text: string) => {
    queueRef.current = queueRef.current.then(async () => {
      await player.speak(text, { speed });
    });
  }, [player, speed]);
  const cancel = useCallback(() => {
    // D-02/Pitfall 5: BOTH reset queue AND stop current source via player.cancel()
    queueRef.current = Promise.resolve();
    player.cancel();
  }, [player]);
  return { enqueue, cancel };
}
```
